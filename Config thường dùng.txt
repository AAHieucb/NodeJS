--> cors:
const whitelist = ["", "https://example2.com"]
app.use(cors({
  credentials: true,
  origin: function (origin, callback) {
    if (whitelist.indexOf(origin) !== -1 || !origin) { // !origin sẽ cho phép các origin undefined truy cập, orgin undefined khi request từ cùng nguồn, nội bộ, hoặc gọi bằng tool client nhu postman
      callback(null, true);
    } else {
      callback(new Error("Not allowed by CORS"))
    }
  },
  origin: ["http://localhost:3000"], // or
  optionsSuccessStatus: 200 // some legacy browsers (IE11, various SmartTVs) choke on 204 nên phải convert về 200 để tương thích
}));

---> origin: "*" cho phép mọi domain, header Access-Control-Allow-Origin trong response cũng là *, nhưng k sử dụng được credentials options. Y hệt app.use(cors()) k options
origin: true tương đương hàm callback luôn trả true thì cho phép mọi domain và header response Access-Control-Allow-Origin sẽ trả đúng từ Origin của request
Tránh nhầm, cái origin là "domain + port + protocol" của nơi host FE, chứ kp của từng client. Nếu chạy local thì nơi host FE và client cùng là máy ta.
---> credentials: true. Nếu client và server khác origin mà client muốn gửi kèm cookie thì fetch sẽ có credentials 'include', khi đó server phải set cors origin được phép gửi kèm và dùng options này để nhận về cookies các thứ.
Nó cũng khiến server gửi response với Access-Control-Allow-Credentials true.

--> compression:
app.use(compression({
  brotli: {
    enabled: true,
    zlib: {
      params: {
        [zlib.constants.BROTLI_PARAM_QUALITY]: 4, // chất lượng nén 0-11, mặc định là 11
        [require('zlib').constants.BROTLI_PARAM_MODE]: require('zlib').constants.BROTLI_MODE_TEXT // mode nén
        [zlib.constants.BROTLI_PARAM_LGWIN]: 22 // kích thước window (10-24)
      }
    }
  },
  level: 6, // Level càng lớn thì compress càng mạnh nhưng nếu lớn quá thì server sẽ phải xử lý nhiều và chậm hơn. Khuyến nghị để mức 6. 
  threshold: 100 *1000, // Đơn vị là byte. tức phải lớn hơn 100kB thì mới thực hiện nén
  filter: (req, res) => { // Filter là điều kiện khi nào thì nén, để mặc định sẽ nén mọi url
    if(req.headers["x-no-compress"]){ // Header k có cái option x-no-compress thì mới nén
      return false;
    }
    return compression.filter(req, res);
  },
}));

--> "post-deploy"
Lần đầu vào thì dùng "pm2 deploy production/staging setup"
